#include <stdio.h>
// Function to calculate the factorial of a number
int factorial(int n) { if (n == 0 || n == 1) return 1; else return n * factorial(n - 1); } int main() { int num = 5; int result;
// Calculate the factorial of num
result = factorial(num);
// Print the result
printf("The factorial of %d is %d\n", num, result); return 0; }
